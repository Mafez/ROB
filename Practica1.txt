PROGRAM LED
	LABEL 3
		SET OUT[4] = 1
		DELAY 50
		SET OUT[4] = 0
		DELAY 50
		GOTO 3
END

PROGRAM PROG1
	GLOBAL SEMG
	DEFINE SEM1
	LABEL 1
		 PRINTLN "ESPERANDO OBJETO EN EL SENSOR 1"
		 WAIT IN[1] = 1
		 PEND SEM1 FROM SEMG
		 
		 IF IN[1] = 0
		 	SET OUT[1] = 1
		 	PRINTLN "ERROR: NO ESTA EL OBJETO EN EL SENSOR 1"
		 	DELAY 200
		 	SET OUT[1] = 0
		 	GOTO 1
		 ENDIF

		 RUN LED
		 PRINTLN "COGIENDO PIEZA DE LA POSICION 2"
		 MOVED 0
		 SPEED 30
		 MOVED 1
		 OPEN
		 SPEED 10
		 MOVED 2
		 CLOSE

		 IF ENC[6] > -100
		 	PRINTLN "NO HE PODIDO COGER EL OBJETO EN LA POSICION 2"
		 	SET OUT[1] = 1
		 	DELAY 500
		 	SET OUT[1] = 0
		 	GOTO 1
		 ENDIF

		 PRITNLN "SOLTANDO PIEZA EN LA POSICION 4"
		 MOVED 1
		 SPEED 30
		 MOVED 3
		 SPEED 10
		 MOVED 4
		 OPEN

		 PRINTLN "VOLVIENDO A LA POSICION INICIAL"
		 MOVED 3
		 SPEED 50
		 MOVED 0
		 CLOSE
		 POST 1 TO SEMG
	GOTO 1
END
